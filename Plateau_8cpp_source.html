103bool Plateau::existeChemin(string villeA, string villeB, Joueur *joueur) const
104{
105 /* Initialisation du tableau distances */
106
107 map<Ville *, int> distance;
108
109 for (size_t i = 0; i < this->getListeVille().size(); i++)
110 {
111 distance[this->getListeVille()[i]] = numeric_limits<int>::max();
112 }

115
116 vector<Ville *> sommets_visites;
117 while (verifDiskstra(this->getListeVille(), distance, sommets_visites))
118 {
119 int distance_minimale = numeric_limits<int>::max();
120 Ville *t = nullptr;
121
122 map<Ville *, int>::iterator it;
123 for (it = distance.begin(); it != distance.end(); it++)
124 {
125 auto recherche = find(sommets_visites.begin(), sommets_visites.end(), it->first);
126
127 if (recherche != sommets_visites.end() && distance[it->first] < distance_minimale)
128 {
129 distance_minimale = distance[it->first];
130 t = it->first;
131 }
132 }
133
134 vector<Ville *> voisins;
135 map<Ville *, int> distanceDuVoisin;
136
137 // Recherche des voisins
138 for (size_t i = 0; i < this->getListeVille().size(); i++)
139 {
140 for (size_t j = 0; j < this->getListeVoieFerree().size(); j++)
141 {
142 if (isVilleInVector(t, this->getListeVoieFerree()[j]->getListeVille()) && isVilleInVector(this->getListeVille()[i], this->getListeVoieFerree()[j]->getListeVille()) && this->getListeVoieFerree()[j]->getJoueur() == joueur)
143 {

146 distanceDuVoisin[this->getListeVille()[i]] = this->getListeVoieFerree()[j]->getPoids();
147 }

150
