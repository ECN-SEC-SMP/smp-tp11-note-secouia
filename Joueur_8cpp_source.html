48 * simplicité d'implémentation l'emporte sur l'optimisation.

50int Joueur::getNbCartes(couleurTrain couleur) const
51{
52 int count = 0;
53 for (const auto &carte : mainCartes)

72/**
73 * @details
74 * Itère une unique fois sur la main et s'arrête dès que `quantite` cartes
75 * ont été retirées. `std::vector::erase` invalide les itérateurs suivants :
76 * on réutilise donc la valeur de retour de `erase`, qui pointe sur l'élément
77 * suivant, pour continuer le parcours sans revalider.
78 */
79bool Joueur::retirerCartes(couleurTrain couleur, int quantite)
80{
81 int count = 0;
82 for (auto it = mainCartes.begin(); it != mainCartes.end() && count < quantite;)
83 {
84 if ((*it)->getCouleurTrain() == couleur)
85 {
86 it = mainCartes.erase(it);
87 count++;
88 }
89 else
90 ++it;
91 }
92 return count == quantite;
93}

95bool Joueur::peutPrendreVoie(couleurTrain couleurVoie, int longueur) const
96{
97 return getNbCartes(couleurVoie) + getNbCartes(couleurTrain::MULTI) >= longueur;

100/**
101 * @details
102 * Stratégie : consommer d'abord les cartes de la couleur demandée (dans la
103 * limite de `longueur`) puis compléter avec des @ref couleurTrain::MULTI.
104 * Les MULTI étant des jokers rares, ils sont ainsi préservés au maximum.
105 */
106bool Joueur::defausserCartesVoie(couleurTrain couleurVoie, int longueur)
107{
108 if (!peutPrendreVoie(couleurVoie, longueur))
109 return false;
110
111 int nbCouleur = std::min(getNbCartes(couleurVoie), longueur);
112 retirerCartes(couleurVoie, nbCouleur);
113 retirerCartes(couleurTrain::MULTI, longueur - nbCouleur);
114 return true;
115}

146/**
147 * @details
148 * Utilise l'idiome *erase-remove* pour retirer `ticket` de @ref missions.
149 * Si le pointeur n'y figure pas, `std::remove` renvoie `missions.end()` et
150 * aucun élément n'est supprimé — la précondition documentée dans le header
151 * empêche toutefois ce cas.
152 */
153bool Joueur::validerTicket(Ticket *ticket, Plateau &plateau)
154{
155 if (!plateau.existeChemin(ticket->getVilleDepart()->getNom(),
156 ticket->getVilleArrivee()->getNom(),
157 this))
158 return false;
159
160 ticketFini++;
161 missions.erase(std::remove(missions.begin(), missions.end(), ticket), missions.end());
162 return true;
163}
