127{
128 ifstream f(fichier);
129 if (!f.is_open()) return;
130 string ligne;
131 getline(f, ligne);
132 while (getline(f, ligne)) {
133 if (ligne.empty() || ligne[0] == '\r') continue;
134 istringstream ss(ligne);

140 voies.push_back(new VoieFerree(
141 vector<Ville*>{getOuCreer(a,villes), getOuCreer(b,villes)},

186/**
187 * @brief Vérifie si deux villes sont connectées par des voies de train appartenant à un joueur
188 *
189 * @param a
190 * @param b
191 * @param voies
192 * @param joueur
193 * @return true
194 * @return false
195 */
196static bool estConnecte(const string& a, const string& b,
197 const vector<VoieFerree*>& voies, Joueur* joueur)
198{
199 if (a == b) return true;
200 map<string, vector<string>> adj;
201 for (auto v : voies) {
202 if (v->getJoueur() != joueur) continue;
203 auto lv = v->getListeVille();
204 for (size_t i = 0; i < lv.size(); i++)
205 for (size_t k = 0; k < lv.size(); k++)
206 if (i != k) adj[lv[i]->getNom()].push_back(lv[k]->getNom());
207 }
208 set<string> vis; vector<string> q = {a}; vis.insert(a);
209 while (!q.empty()) {
210 string cur = q.back(); q.pop_back();
211 if (cur == b) return true;
212 for (const auto& n : adj[cur])
213 if (!vis.count(n)) { vis.insert(n); q.push_back(n); }
214 }
215 return false;
216}

242 vector<Ticket*> locale;

245 /**
246 * @brief Construct a new Pioche Tickets object
247 *
248 * @param n
249 * @param g
250 */
251 PiocheTickets(int n, mt19937& g) : partieLeft(n), rng(g) {}
252 /**
253 * @brief Vérifie si la pioche est vide
254 *
255 * @return true
256 * @return false
257 */
258 bool vide() const { return partieLeft <= 0 && locale.empty() && defausse.empty(); }
259 /**
260 * @brief Pioche un ticket pour le joueur
261 *
262 * @param j
263 * @param partie
264 * @param log
265 * @return true
266 * @return false
267 */
268 bool piocher(Joueur& j, Partie& partie, vector<string>& log)
269 {
270 if (partieLeft <= 0 && locale.empty()) {
271 if (defausse.empty()) { log.push_back("Pioche tickets vide !"); return false; }
272 locale = defausse; defausse.clear();
273 shuffle(locale.begin(), locale.end(), rng);
274 log.push_back("[Défausse recyclée]");
275 }
276 if (partieLeft > 0) {
277 size_t avant = j.getMissions().size();
278 partie.piocherTicket(j); partieLeft--;
279 if (j.getMissions().size() > avant) {
280 auto t = j.getMissions().back();
281 log.push_back("Ticket: " + t->getVilleDepart()->getNom()
282 + " > " + t->getVilleArrivee()->getNom());
283 }
284 } else {
285 Ticket* t = locale.back(); locale.pop_back();
286 j.ajouterTicket(t);
287 log.push_back("Ticket: " + t->getVilleDepart()->getNom()
288 + " > " + t->getVilleArrivee()->getNom());
289 }
290 return true;
291 }
292 /**
293 * @brief Défausse des tickets
294 *
295 * @param tickets
296 */
297 void defausser(vector<Ticket*>& tickets)
298 { for (auto t : tickets) defausse.push_back(t); }
299};

302 *
303 * @param j
304 * @param voies
305 * @param plateau
306 * @param pioche
307 * @param partie
308 * @param log

310static void validerTickets(Joueur* j, const vector<VoieFerree*>& voies,
311 Plateau* plateau, PiocheTickets& pioche,
312 Partie& partie, vector<string>& log)
313{
314 vector<Ticket*> ms = j->getMissions();

321 pioche.piocher(*j, partie, log);

568 *

576static void drawThickLine(QPainter& p,
577 QPointF a, QPointF b,
578 QColor color, float thick, float perpOff = 0.f)
579{
580 QPointF d = b - a;

582 if (len < 1.0) return;
583 double nx = -d.y() / len, ny = d.x() / len;
584 QPointF start = QPointF(a.x() + nx * perpOff, a.y() + ny * perpOff);

587 p.save();
588 p.translate(start);
589 p.rotate(angleDeg);
590 p.fillRect(QRectF(0, -thick/2.0, len, thick), color);
591 p.restore();
592}
593/**
594 * @brief Calcule la distance d'un point à un segment
595 *
596 * @param pt
597 * @param a
598 * @param b
599 * @return double
600 */
601static double distSeg(QPointF pt, QPointF a, QPointF b)
602{
603 QPointF ab = b - a, ap = pt - a;
604 double t = (ap.x()*ab.x() + ap.y()*ab.y()) / (ab.x()*ab.x() + ab.y()*ab.y() + 1e-6);
605 t = qBound(0.0, t, 1.0);
606 QPointF cl = a + ab * t, dv = pt - cl;
607 return std::sqrt(dv.x()*dv.x() + dv.y()*dv.y());
608}
609/**
610 * @brief Calcule les offsets pour les voies parallèles
611 *
612 * @param voies
613 * @return map<VoieFerree*, float>
614 */
615static map<VoieFerree*, float> calcOffsets(const vector<VoieFerree*>& voies)
616{
617 map<VoieFerree*, float> offsets;
618 map<pair<string,string>, vector<VoieFerree*>> grp;
619 for (auto v : voies) {
620 auto lv = v->getListeVille();
621 string a = lv[0]->getNom(), b = lv[1]->getNom();
622 if (a > b) swap(a, b);
623 grp[{a, b}].push_back(v);
624 }
625 for (auto& kv : grp) {
626 auto& g = kv.second;
627 if (g.size() == 1) { offsets[g[0]] = 0.f; continue; }
628 float start = -(float(g.size()) - 1.f) * 7.f / 2.f;
629 for (size_t i = 0; i < g.size(); i++)
630 offsets[g[i]] = start + float(i) * 7.f;
631 }
632 return offsets;
633}

1317/**
1318 * @brief Trouve la voie ferée la plus proche d'un point donné, dans un rayon de 14 pixels. Ignore les voies déjà prises ou mal positionnées.
1319 *
1320 * @param mp
1321 * @param gs
1322 * @return VoieFerree*
1323 */
1324static VoieFerree* findRoute(QPointF mp, const GS& gs)
1325{
1326 VoieFerree* best = nullptr;
1327 double bestD = 14.0;
1328 for (auto v : gs.voies) {
1329 if (v->getJoueur()) continue;
1330 auto lv = v->getListeVille();
1331 if (!CITY_POS.count(lv[0]->getNom()) || !CITY_POS.count(lv[1]->getNom())) continue;
1332 QPointF pa = CITY_POS.at(lv[0]->getNom());
1333 QPointF pb = CITY_POS.at(lv[1]->getNom());
1334 float off = gs.offsets.count(v) ? gs.offsets.at(v) : 0.f;
1335 QPointF d = pb - pa;
1336 double len = std::sqrt(d.x()*d.x() + d.y()*d.y());
1337 QPointF perp = len > 0 ? QPointF(-d.y()/len, d.x()/len) : QPointF(0,0);
1338 double dist = distSeg(mp, pa + perp*off, pb + perp*off);
1339 if (dist < bestD) { bestD = dist; best = v; }
1340 }
1341 return best;
1342}
